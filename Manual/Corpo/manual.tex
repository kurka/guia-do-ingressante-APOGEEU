    \colorbox{titlegrey}{\vbox to5mm{\hsize111mm\null\vfill

        \textbf{#2}\addcontentsline{toc}{chapter}{#2}\par\smallskip}}%
  \par\vspace{24pt}\parskip 7.2pt\raggedright
}{% end story
